字符串是一个字符序列，字符串最左端位置标记为0，依次增加，字符串中的编号叫索引
a = "0123456789"
a[1] = '1'
a[1:7] = '123456'  （前一个是开始，后一个是结束，取不到结束）
a[8:1:-1]  = '8765432'  (正负表方向，数字表示间隔) a[ : :-2] = '97531'
